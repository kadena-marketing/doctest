import React, { useState } from 'react';

export function CollapsibleSection({ title, children }) {
  const [isOpen, setIsOpen] = useState(false);

  return (
    <div>
      <h3 onClick={() => setIsOpen(!isOpen)} style={{ cursor: 'pointer', marginBottom: '0.5em', color: 'var(--k-secondary)' }}>
        {title} {isOpen ? '▼' : '▶'}
      </h3>
      {isOpen && <div style={{ marginBottom: '1em' }}>{children}</div>}
    </div>
  );
}

export function InteractiveTable() {
  const [sortDirection, setSortDirection] = useState('asc');
  const [data, setData] = useState([
    { category: 'Transaction Costs', amount: 'Low' },
    { category: 'Developer Grants', amount: 'High' }
  ]);

  const sortData = () => {
    const sortedData = [...data].sort((a, b) => {
      if (sortDirection === 'asc') {
        return a.amount.localeCompare(b.amount);
      } else {
        return b.amount.localeCompare(a.amount);
      }
    });
    setSortDirection(sortDirection === 'asc' ? 'desc' : 'asc');
    setData(sortedData);
  };

  return (
    <div>
      <button onClick={sortData} style={{ marginBottom: '1em' }}>
        Sort by Amount ({sortDirection})
      </button>
      <table style={{ width: '100%', borderCollapse: 'collapse' }}>
        <thead>
          <tr>
            <th style={{ border: '1px solid #ddd', padding: '8px' }}>Category</th>
            <th style={{ border: '1px solid #ddd', padding: '8px' }}>Amount</th>
          </tr>
        </thead>
        <tbody>
          {data.map((item, index) => (
            <tr key={index}>
              <td style={{ border: '1px solid #ddd', padding: '8px' }}>{item.category}</td>
              <td style={{ border: '1px solid #ddd', padding: '8px' }}>{item.amount}</td>
            </tr>
          ))}
        </tbody>
      </table>
    </div>
  );
}

## Key Highlights

### **The Gas Station: A Game-Changer for Enterprises**
We’re proud to introduce the **Crypto Gas Station**, a pioneering feature on the Kadena blockchain. This breakthrough allows **decentralized applications (dApps)** and enterprise solutions to completely absorb transaction costs, presenting a seamless and **cost-free experience** for end users. The Crypto Gas Station is designed to simplify the user experience and encourage adoption by removing transactional friction. Enterprises can integrate this feature into their applications to cover gas fees on behalf of their users, thereby eliminating the need for users to handle tokens and reducing barriers to entry.
